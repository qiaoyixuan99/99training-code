    const STORAGE_COLLECT = "yummy_collected_ids";
    const STORAGE_LAST_DATE = "yummy_last_lottery_date";

    let collectedIds = [];      // 已经收集的菜品id数组
    let todayHasDrawn = false;  // 今日是否已经抽奖过（用于每日首次必得新菜品）

    // 辅助函数
    function getTodayStr() {
        const d = new Date();
        return `${d.getFullYear()}-${d.getMonth()+1}-${d.getDate()}`;
    }

    // 加载收藏
    function loadCollection() {
        const stored = localStorage.getItem(STORAGE_COLLECT);
        if (stored) {
            try {
                collectedIds = JSON.parse(stored);
                if (!Array.isArray(collectedIds)) collectedIds = [];
            } catch(e) { collectedIds = []; }
        }
        // 如果没有任何收藏，赠送一个基础菜品让柜子不空 (增强新手体验)
        if (collectedIds.length === 0) {
            const starter = MENU[0].id;
            collectedIds = [starter];
            saveCollection();
        }
        // 去除重复
        collectedIds = [...new Set(collectedIds)];
        saveCollection();
        renderCabinet();
    }

    function saveCollection() {
        localStorage.setItem(STORAGE_COLLECT, JSON.stringify(collectedIds));
    }

    // 检查今日抽奖状态
    function checkDailyStatus() {
        const lastDate = localStorage.getItem(STORAGE_LAST_DATE);
        const today = getTodayStr();
        todayHasDrawn = (lastDate === today);
        if (todayHasDrawn) {
            rewardMsgDiv.innerHTML = "🍬 今日已抽过奖啦！明天再来拿新菜品吧 🍬";
        } else {
            rewardMsgDiv.innerHTML = "⭐ 每天首次抽奖必得新菜品！解锁后放入柜子 ⭐";
        }
    }

    function markTodayDrawn() {
        localStorage.setItem(STORAGE_LAST_DATE, getTodayStr());
        todayHasDrawn = true;
        rewardMsgDiv.innerHTML = "✅ 今日福利已领！明日继续收集新料理 ✅";
    }

    // 渲染收藏柜（精美架子风格）
    function renderCabinet() {
        if (!cabinetShelf) return;
        const collectedDishes = collectedIds.map(id => MENU.find(d => d.id === id)).filter(d => d);
        if (collectedDishes.length === 0) {
            cabinetShelf.innerHTML = `<div class="empty-cabinet">🍽️ 柜子上空空如也... 快抽奖收集吧 🍽️</div>`;
            return;
        }
        cabinetShelf.innerHTML = "";
        collectedDishes.forEach(dish => {
            const card = document.createElement("div");
            card.className = "collect-item";
            card.innerHTML = `
                <div class="collect-img">
                    <img src="${dish.img}" alt="${dish.name}" loading="lazy" onerror="this.src='https://cdn-icons-png.flaticon.com/512/1998/1998592.png'">
                </div>
                <div class="collect-name">${dish.name}</div>
            `;
            cabinetShelf.appendChild(card);
        });
    }

    // 添加新菜品到收藏 (如果还不存在)
    function addToCollection(dishId) {
        if (!collectedIds.includes(dishId)) {
            collectedIds.push(dishId);
            saveCollection();
            renderCabinet();
            return true; // 新收集
        }
        return false;
    }

    // 显示当前抽中的菜品 (主页面大图)
    function displayCurrentDish(dish) {
        dishImage.src = dish.img;
        dishImage.alt = dish.name;
        currentDishNameSpan.innerText = dish.name;
        // 图片加载备用
        dishImage.onerror = function() {
            this.src = "https://cdn-icons-png.flaticon.com/512/1998/1998592.png";
        };
    }

    // 抽奖核心逻辑
    function performLottery() {
        const allIds = MENU.map(d => d.id);
        const isFullCollection = collectedIds.length === allIds.length;
        let selectedDish = null;
        let isNew = false;
        let msgExtra = "";

        // 奖励机制: 今日首次抽奖 且 图鉴未满 → 必得未收集菜品
        if (!todayHasDrawn && !isFullCollection) {
            const uncollected = allIds.filter(id => !collectedIds.includes(id));
            if (uncollected.length > 0) {
                const newId = uncollected[Math.floor(Math.random() * uncollected.length)];
                selectedDish = MENU.find(d => d.id === newId);
                isNew = addToCollection(selectedDish.id);
                msgExtra = " 🎉 新料理上柜！🎉";
                markTodayDrawn();
            } else {
                // 理论上有uncollected长度>0不会进这里，兜底
                selectedDish = MENU[Math.floor(Math.random() * MENU.length)];
                isNew = addToCollection(selectedDish.id);
                if(isNew) msgExtra = " ✨ 新收藏 ✨";
                markTodayDrawn();
            }
        } 
        else {
            // 普通抽奖 (今天非首次 或者 已全收集)
            selectedDish = MENU[Math.floor(Math.random() * MENU.length)];
            const wasNew = addToCollection(selectedDish.id);
            if (wasNew) {
                msgExtra = " 🌟 意外收获！新菜品进柜 🌟";
            } else {
                msgExtra = " 🍽️ 回味经典，再次品尝 🍽️";
            }
            if (!todayHasDrawn && isFullCollection) {
                // 全收集情况下首次抽奖
                markTodayDrawn();
                msgExtra = " 🏆 美食大师！全图鉴已解锁 🏆";
            }
            if(todayHasDrawn) {
                msgExtra = " 🍴 今日复抽，美味依旧 🍴";
            }
        }

        if (selectedDish) {
            displayCurrentDish(selectedDish);
            // 额外更新提示文案 (保留奖励机制反馈)
            if (!todayHasDrawn && !isFullCollection && msgExtra.includes("新料理")) {
                rewardMsgDiv.innerHTML = `✨ 每日福利：获得 ${selectedDish.name} ✨${msgExtra}`;
            } else if (msgExtra) {
                rewardMsgDiv.innerHTML = `${msgExtra} ${todayHasDrawn ? '明天再来领新菜品哦~' : '好运连连'}`;
            } else {
                rewardMsgDiv.innerHTML = `🎲 今日大餐：${selectedDish.name} 🎲`;
            }
            // 如果抽奖后全收集则更新消息
            if(collectedIds.length === allIds.length) {
                rewardMsgDiv.innerHTML = "🏅 恭喜集齐全套美食！每日都可以快乐抽奖 🏅";
            }
        } else {
            // fallback 极少数情况
            const fallback = MENU[0];
            displayCurrentDish(fallback);
            rewardMsgDiv.innerHTML = "🍬 开饭啦！再试一次～ 🍬";
        }

        // 刷新柜子展示
        renderCabinet();
        // 同步再次检查今日抽奖标志以便更新
        checkDailyStatus();   // 更新message但不要覆盖掉奖励反馈太生硬，但保留好
        if (!todayHasDrawn && !isFullCollection && !selectedDish) {
            // 保险
        } else {
            if(todayHasDrawn && rewardMsgDiv.innerHTML.includes("每日首次抽奖")){
                // 刚抽完后todayHasDrawn为true，修正文案美观
                if(collectedIds.length === allIds.length){
                    rewardMsgDiv.innerHTML = "🍱 全图鉴达成！尽情享受美食盛宴 🍱";
                } else if(!rewardMsgDiv.innerHTML.includes("新料理")){
                    rewardMsgDiv.innerHTML = "🎈 抽奖完成！看看你的美食柜～ 🎈";
                }
            }
        }
    }

    // 初始化默认展示 (根据第一个收藏菜品 或默认披萨)

        collectionTabBtn.classList.add("active-tab");
        lotteryTabBtn.classList.remove("active-tab");
        // 每次进入收藏柜刷新柜子渲染 (保证最新收集)
        renderCabinet();
    }

    // 绑定事件
    function bindEvents() {
        lotteryBtn.addEventListener("click", () => {
            performLottery();
        });
        lotteryTabBtn.addEventListener("click", switchToLottery);
        collectionTabBtn.addEventListener("click", switchToCollection);
    }

    // 首次初始化
    function init() {
        loadCollection();
        checkDailyStatus();
        initDefaultDisplay();
        bindEvents();
        renderCabinet();
    }
